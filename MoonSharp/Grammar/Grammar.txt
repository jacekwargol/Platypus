expression =	literal | grouping | unary | binary | operator ;

literal =		INTEGER | FLOAT | STRING | "true" | "false" ;
grouping =		"(" expression ")" ;
unary =			("-" | "!") expression ;
binary =		expression operator expression ;
operator =		"." | "+" | "-" | "=" 
				| "==" | "!=" | "<=" | ">="
				| "<" | ">"
;
